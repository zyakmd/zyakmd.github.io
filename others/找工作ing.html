//5.0分制——绩点计算器

//低于60分绩点记为0

//2019.04.30 ——LeoHao

#include<bits/stdc++.h>

using namespace std;

int SubjectNum=0;           //学科数目

float *score=NULL;          //学科分数

float *credit=NULL;         //学科学分

float *gradePoint=NULL;     //学科绩点

bool *isProfessional=NULL;  //是否专业课

float wholeCredit=0;        //专业课总学分

float GPA=0;                //绩点

void ScoreToPoint();        //成绩转绩点

void GetGPA();              //计算GPA

int main()

{

    int i=0;

    cout<<"nums:"<<endl;

    cin>>SubjectNum;

    score=new float[SubjectNum];

    credit=new float[SubjectNum];

    gradePoint=new float[SubjectNum];

    isProfessional=new bool[SubjectNum];

    // 门科目的成绩，并标注是否为专业课。（例：算法设计 4个学分,95分,为专业课,则输入1 4 95,不为专业课,则输入0 4 95)

    cout<<"input "<<SubjectNum<<":1 Credits Score"<<endl;

    for(i=0;i<SubjectNum;i++)

        cin>>isProfessional[i]>>credit[i]>>score[i];

                            //计算专业课总学分

    for(i=0;i<SubjectNum;i++)

        wholeCredit+=(isProfessional[i]*credit[i]);

    ScoreToPoint();         //成绩转绩点

    GetGPA();               //计算GPA

    cout<<"every:"<<endl<<endl;

    for(i=0;i<SubjectNum;i++)

        cout<<"each"<<i+1<<"\t";

    cout<<endl;

    for(i=0;i<SubjectNum;i++)

        cout<<gradePoint[i]<<"\t";  

    cout<<endl<<endl;

    cout<<"In sum, GPA:"<<GPA<<endl;

    getchar();

    getchar();

    return 0;

}

void ScoreToPoint()         //成绩转绩点

{

    int i=0;

    for(i=0;i<SubjectNum;i++)

    {

        if(score[i]<60)

        gradePoint[i]=0;    //低于60分绩点记为0

        else

        gradePoint[i]=5.0-0.1*(100-score[i]);

    }

}

void GetGPA()               //计算GPA

{

    int i=0;

    for(i=0;i<SubjectNum;i++)

        GPA+=(isProfessional[i]*gradePoint[i]*(credit[i]/wholeCredit));

}

本人完成工作：

  • 实现 自定义传输协议数据格式，保证了数据的完整性和一致性，解决粘包、半包等问题。

项目描述：项目来源于校内金种子课题，主要实现了用户登录注册管理、发布商品、点赞收藏及系统通知、下单等功能，实现了一个高可用、高并发、高性能平台。

本人完成工作：

  • 实现 统一的异常处理，日志记录。
  • 使用 EasyExcel + 线程池 提高数据批量导入效率，导入数据的时间从 6min 优化至 21s。
  • 使用 Redis 实现高频数据缓存，提高并发访问能力，数据范围查询速度从 1.5s 优化至 7ms。
  • 实现 自定义 Redis 序列化器来解决数据乱码的问题，外加 Spring Scheduler 定时任务来实现缓存预热。
  • 使用 互斥锁和布隆过滤器解决缓存击穿、缓存穿透的问题，设置缓存时间波动避免缓存雪崩的问题。
  • 使用 乐观锁 + Redis 进行库存扣减，避免超卖。利用分布式锁保证数据一致性。

  • 使用 RocketMQ 进行流量削峰，实现了点赞，关注，收藏等通知的异步处理。
  • 使用 Nginx 实现负载均衡，降低单个服务器压力，保证服务的高可用。部署静态资源，提高访问性能。
